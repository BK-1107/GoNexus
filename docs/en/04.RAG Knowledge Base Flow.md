# 04. RAG Knowledge Base Flow

The RAG knowledge base converts uploaded documents into chunks, embeds them, and stores them in Redis Stack.
When chat uses the RAG model, the backend retrieves relevant chunks from Redis, builds a prompt, and calls the LLM.

## 1. Full Flow

```text
KnowledgeBase.tsx
  -> fileApi.upload
  -> POST /api/v1/file/upload
  -> Router
  -> JWT middleware
  -> controller
  -> service saves to uploads/{username}
  -> RAGIndexer
  -> extractPlainText
  -> splitText
  -> embedding
  -> Redis Stack vector index

When chatting with RAG model:
AIHelper
  -> AliRAGModel
  -> RAGQuery
  -> retrieve relevant chunks from Redis
  -> BuildRAGPrompt
  -> LLM
```

```text
GoNexus/frontend/src/pages/KnowledgePage.tsx
```

```go
func UploadRagFile(c *gin.Context) {
	res := new(UploadFileResponse)

	// Read file field from multipart form
	uploadedFile, err := c.FormFile("file")
	if err != nil {
		c.JSON(http.StatusOK, res.CodeOf(code.CodeInvalidParams))
		return
	}

	// Get current user from JWT middleware context
	username := c.GetString("userName")
	if username == "" {
		c.JSON(http.StatusOK, res.CodeOf(code.CodeInvalidToken))
		return
	}

	// Let service validate file, save it, and build RAG index
	filePath, err := file.UploadRagFile(username, uploadedFile)
	if err != nil {
		c.JSON(http.StatusOK, res.CodeOf(code.CodeServerBusy))
		return
	}

	// Return saved file path
	res.Success()
	res.FilePath = filePath
	c.JSON(http.StatusOK, res)
}
```

## 5. Service: Save to uploads/{username}

File: `GoNexus/service/file/file.go`

```go
func UploadRagFile(username string, file *multipart.FileHeader) (string, error) {
	// Validate file type and filename
	if err := utils.ValidateFile(file); err != nil {
		return "", err
	}

	// Create user directory: uploads/{username}
	userDir := filepath.Join("uploads", username)
	if err := os.MkdirAll(userDir, 0755); err != nil {
		return "", err
	}

	// One shared knowledge-base index per user
	indexID := rag.UserKnowledgeIndexID(username)

	// Generate server-side filename with UUID
	uuid := utils.GenerateUUID()
	ext := filepath.Ext(file.Filename)
	filename := uuid + ext
	filePath := filepath.Join(userDir, filename)
```

```go
// Open uploaded file
src, err := file.Open()

defer src.Close()

// Create destination file
dst, err := os.Create(filePath)

	return "", err
}
defer dst.Close()

	return "", err
```

## 8. RAG: Read File and Split Chunks

Files: `GoNexus/common/rag/rag.go`, `GoNexus/common/rag/split.go`

```go
func (r *RAGIndexer) IndexFile(ctx context.Context, filePath string) error {
	// Read txt/md text content
	content, err := extractPlainText(filePath)
	if err != nil {
		return err
	}

	// Split into chunks
	chunks := splitText(content, defaultChunkSize, defaultChunkOverlap)
	if len(chunks) == 0 {
		return fmt.Errorf("file has no indexable content: %s", filePath)
	}
```

```go
for i, chunk := range chunks {
	docs = append(docs, &schema.Document{
		// One document ID for each chunk
		ID: fmt.Sprintf("%s_chunk_%d", fileID, i),

		// Chunk text
		Content: chunk,

		// Source file and chunk number
		MetaData: map[string]any{
			"source": filePath,
			"chunk":  i,
		},
	})
}

// Store into Redis vector index
_, err = r.indexer.Store(ctx, docs)
return err
```

```go
	// Overlap context between adjacent chunks
```

```go
func (o *AliRAGModel) StreamResponse(ctx context.Context, messages []*schema.Message, cb StreamCallback) (string, error) {
	// Get latest user question
	query := messages[len(messages)-1].Content

	// Create RAG query object for current user
	ragQuery, err := rag.NewRAGQuery(ctx, o.username)
	if err != nil {
		return o.streamWithoutRAG(ctx, messages, cb)
	}

	// Retrieve relevant document chunks from Redis
	docs, err := ragQuery.RetrieveDocuments(ctx, query)
	if err != nil {
		return o.streamWithoutRAG(ctx, messages, cb)
	}

	// Build final RAG prompt
	ragPrompt := rag.BuildRAGPrompt(query, docs)
```
