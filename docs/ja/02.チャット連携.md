# 02. チャット連携

この章では、現在のフロントエンドの主経路である SSE ストリーミングチャットを扱います。

## 1. 全体の流れ

```text
ChatInput.tsx
  -> useStreaming.ts
  -> POST /api/v1/AI/chat/send-stream または /send-stream-new-session
  -> Router
  -> JWT middleware
  -> controller
  -> service
  -> AIHelperManager
  -> AIModelFactory
  -> AIHelper
  -> AIModel.StreamResponse
  -> SSE data: xxx\n\n をフロントエンドへ返す
  -> frontend updateLastMessage
  -> RabbitMQ
  -> dao/message
  -> MySQL
```

対応するファイル：

```text
GoNexus/frontend/src/components/chat/ChatInput.tsx
  -> GoNexus/frontend/src/hooks/useStreaming.ts
  -> GoNexus/router/AI.go
  -> GoNexus/middleware/jwt/jwt.go
  -> GoNexus/controller/session/session.go
  -> GoNexus/service/session/session.go
  -> GoNexus/common/aihelper/manager.go
  -> GoNexus/common/aihelper/factory.go
  -> GoNexus/common/aihelper/aihelper.go
  -> GoNexus/common/aihelper/model.go
  -> GoNexus/common/rabbitmq/meesage.go
  -> GoNexus/dao/message/message.go
  -> GoNexus/model/message.go
```

## 2. POST：フロントエンドがストリーミング API を選ぶ

ファイル：`GoNexus/frontend/src/components/chat/ChatInput.tsx`

```tsx
// currentSessionId があれば既存会話、なければ新規会話
const url = currentSessionId 
  ? "/AI/chat/send-stream" 
  : "/AI/chat/send-stream-new-session"

// question はユーザー入力、modelType は選択されたモデル種別
const body: any = { 
  question: messageContent, 
  modelType: modelType 
}

// 既存会話では sessionId も送る
if (currentSessionId) {
  body.sessionId = currentSessionId
}

// SSE ストリーミングリクエストへ進む
await startStream(url, body)
```

## 4. Router：チャット API にマッチする

ファイル：`GoNexus/router/AI.go`

```go
func AIRouter(r *gin.RouterGroup) {
	{
		// 新規会話：session を作成してから AI 回答をストリーミングする
		r.POST("/chat/send-stream-new-session", session.CreateStreamSessionAndSendMessage)

		// 既存会話：指定 session で会話を継続する
		r.POST("/chat/send-stream", session.ChatStreamSend)
	}
}
```

## 5. JWT：token を検証して userName を設定する

ファイル：`GoNexus/router/router.go`、`GoNexus/middleware/jwt/jwt.go`

```go
// /AI 以下の API はログインが必要
AIGroup := enterRouter.Group("/AI")
AIGroup.Use(jwt.Auth())
AIRouter(AIGroup)
```

```go
// Authorization ヘッダーから token を読む
authHeader := c.GetHeader("Authorization")
if authHeader != "" && strings.HasPrefix(authHeader, "Bearer ") {
	token = strings.TrimPrefix(authHeader, "Bearer ")
}

// token を解析して現在ユーザーを取得する
userName, ok := myjwt.ParseToken(token)
if !ok {
	c.JSON(http.StatusOK, res.CodeOf(code.CodeInvalidToken))
	c.Abort()
	return
}

// 後続 controller 用に userName を Gin context へ保存する
c.Set("userName", userName)
c.Next()
```

## 6. Controller：新規会話ストリーミングチャット

ファイル：`GoNexus/controller/session/session.go`

```go
func CreateStreamSessionAndSendMessage(c *gin.Context) {
	req := new(CreateSessionAndSendMessageRequest)

	// JWT middleware が保存した現在ユーザーを取得する
	userName := c.GetString("userName")

	// question と modelType をバインドする
	if err := c.ShouldBindJSON(req); err != nil {
		c.JSON(http.StatusOK, gin.H{"error": "Invalid parameters"})
		return
	}
```

```go
// ブラウザへ SSE レスポンスであることを伝える
c.Header("Content-Type", "text/event-stream")
c.Header("Cache-Control", "no-cache")
c.Header("Connection", "keep-alive")
c.Header("Access-Control-Allow-Origin", "*")
c.Header("X-Accel-Buffering", "no")
```

```go
// 新規チャットでは先に session を作成する
sessionID, code_ := session.CreateStreamSessionOnly(userName, req.UserQuestion)
if code_ != code.CodeSuccess {
	c.SSEvent("error", gin.H{"message": "Failed to create session"})
	return
}

// 生成した sessionId を先にフロントエンドへ送る
c.Writer.WriteString(fmt.Sprintf("data: {\"sessionId\": \"%s\"}\n\n", sessionID))
c.Writer.Flush()
```

```go
// 実際のストリーミングチャット処理へ進む
code_ = session.StreamMessageToExistingSession(
	userName,
	sessionID,
	req.UserQuestion,
	req.ModelType,
	http.ResponseWriter(c.Writer),
)
```

```go

cb := func(msg string) {
	_, err := writer.Write([]byte("data: " + msg + "\n\n"))
	if err != nil {
		return
	}
	flusher.Flush()
}
```

```go

_, err_ := helper.StreamResponse(userName, ctx, cb, userQuestion)
if err_ != nil {
	return code.AIModelFail
}

// ストリーム終了を通知する

flusher.Flush()
```

```ts
updateLastMessage: (content) => set((state) => {
  const newMessages = [...state.messages]
  if (newMessages.length > 0) {
    // 最後のメッセージは事前に挿入した空の AI メッセージ
    newMessages[newMessages.length - 1] = { 
      ...newMessages[newMessages.length - 1], 
      // 最新の累積内容で置き換える
      content 
    }
  }
  return { messages: newMessages }
})
```
